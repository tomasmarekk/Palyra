Create a browser session

Usage: palyra browser session create [OPTIONS]

Options:
      --principal <PRINCIPAL>
      --channel <CHANNEL>
      --idle-ttl-ms <IDLE_TTL_MS>
      --allow-private-targets
      --allow-downloads
      --allow-domain <ACTION_ALLOWED_DOMAINS>
      --persistence-enabled
      --persistence-id <PERSISTENCE_ID>
      --profile-id <PROFILE_ID>
      --private-profile
  -h, --help                                   Print help

Global Options:
      --profile <PROFILE>
      --config <CONFIG_PATH>
      --state-root <STATE_ROOT>
  -v, --verbose...                     Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>          [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>  [default: text] [possible values: text, json, ndjson]
      --plain
      --no-color
