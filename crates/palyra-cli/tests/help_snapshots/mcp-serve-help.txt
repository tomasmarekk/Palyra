Run the stdio MCP server

Usage: palyra mcp serve [OPTIONS]

Options:
      --grpc-url <GRPC_URL>
      --token <TOKEN>
      --principal <PRINCIPAL>
      --device-id <DEVICE_ID>
      --channel <CHANNEL>
      --session-key <SESSION_KEY>
      --session-label <SESSION_LABEL>
      --require-existing
      --reset-session
      --read-only
      --allow-sensitive-tools
  -h, --help                           Print help

Global Options:
      --profile <PROFILE>

      --expect-profile <EXPECT_PROFILE>

      --config <CONFIG_PATH>

      --state-root <STATE_ROOT>

  -v, --verbose...
          Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>
          [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>
          [default: text] [possible values: text, json, ndjson]
      --plain

      --no-color

      --allow-profile-mismatch

      --allow-strict-profile-actions
