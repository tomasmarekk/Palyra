Open a tab in a browser session

Usage: palyra browser tabs <SESSION_ID> open [OPTIONS] --url <URL>

Options:
      --url <URL>
      --activate
      --timeout-ms <TIMEOUT_MS>
      --allow-redirects
      --max-redirects <MAX_REDIRECTS>
      --allow-private-targets
  -h, --help                           Print help

Global Options:
      --profile <PROFILE>
      --config <CONFIG_PATH>
      --state-root <STATE_ROOT>
  -v, --verbose...                     Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>          [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>  [default: text] [possible values: text, json, ndjson]
      --plain
      --no-color
