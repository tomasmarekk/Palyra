Observe the active tab and emit a bounded page snapshot

Usage: palyra browser snapshot [OPTIONS] <SESSION_ID>

Arguments:
  <SESSION_ID>

Options:
      --include-dom-snapshot
      --include-accessibility-tree
      --include-visible-text
      --max-dom-snapshot-bytes <MAX_DOM_SNAPSHOT_BYTES>
      --max-accessibility-tree-bytes <MAX_ACCESSIBILITY_TREE_BYTES>
      --max-visible-text-bytes <MAX_VISIBLE_TEXT_BYTES>
      --output <OUTPUT>
  -h, --help                                                         Print help

Global Options:
      --profile <PROFILE>
      --config <CONFIG_PATH>
      --state-root <STATE_ROOT>
  -v, --verbose...                     Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>          [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>  [default: text] [possible values: text, json, ndjson]
      --plain
      --no-color
